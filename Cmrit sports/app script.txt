function doPost(e) {
  var ss = SpreadsheetApp.openByUrl('https://docs.google.com/spreadsheets/d/1S6SSnSFd2EVNhp5_IzKIlmxc7CGjksQbQqrlPm1veLo/edit?gid=0#gid=0'); // Replace 'YOUR_SPREADSHEET_ID' with the ID of your Google Sheet
  var sheet = ss.getSheetByName('cricket'); // Replace 'Sheet1' with the name of your sheet
 
  var row = [];
  row.push(new Number()); // Timestamp
  row.push(e.parameter.name);
  row.push(e.parameter.number);
  row.push(e.parameter.regnumber);
  row.push(e.parameter.age);
  row.push(e.parameter.gender);
  row.push(e.parameter.field);
 
  sheet.appendRow(row);
 
  return HtmlService.createHtmlOutputFromFile("thanku");
}
